``` sql
SELECT first_name, last_name, salary, hire_date
FROM hr.employees
ORDER BY salary DESC;

-- probar join
SELECT e.first_name, e.last_name, d.department_name
FROM hr.employees e
JOIN hr.departments d ON e.department_id = d.department_id
WHERE d.department_name = 'IT';
```

Realemente es muy sql puro que lo bueno esta en las declaraciones ya vistas antes(if, while, etc)

``` sql
DECLARE
    v_conteo NUMBER;
    v_dept_nombre VARCHAR2(50) := 'IT';
BEGIN
    -- Buscamos cuántos empleados hay en el departamento 'IT'
    SELECT COUNT(*) 
    INTO v_conteo 
    FROM hr.employees e
    JOIN hr.departments d ON e.department_id = d.department_id
    WHERE d.department_name = v_dept_nombre;

    -- Lógica condicional (IF-THEN-ELSE)
    IF v_conteo > 0 THEN
        DBMS_OUTPUT.PUT_LINE('El departamento ' || v_dept_nombre || ' tiene ' || v_conteo || ' empleados activos.');
    ELSE
        DBMS_OUTPUT.PUT_LINE('No se encontraron empleados en el departamento ' || v_dept_nombre || '.');
    END IF;
EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Ocurrió un error inesperado.');
END;

```
